## Install

```bash
npm install --save pendulum
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'pendulum'
import 'pendulum/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
